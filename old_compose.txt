services:
  web:
    build: .
    container_name: my-web-app-container
    command: bash -c "sleep 10 && . /APP_DIR/venv/bin/activate && python /APP_DIR/app/manage.py migrate && python /APP_DIR/app/manage.py runserver localhost:8000"
    ports:
      - "8000:8000"
    environment:
      - PYTHONUNBUFFERED=1
    volumes:
      - .:/APP_DIR
    networks:
      - webnet
    depends_on:
      - db

  db:
    image: mysql:8.0
    container_name: mysql-db
    environment:
      MYSQL_ROOT_PASSWORD: rootpassword
      MYSQL_DATABASE: mydb
      MYSQL_USER: user
      MYSQL_PASSWORD: password
    volumes:
      - db_data:/var/lib/mysql
    networks:
      - webnet

  nginx:
    image: nginx:latest
    container_name: nginx
    ports:
      - "8080:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
    depends_on:
      - web
networks:
  webnet:

volumes:
  db_data:
